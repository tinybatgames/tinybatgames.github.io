// Firebase Console > Firestore Database > Rules sekmesine yapıştır, "Publish" bas.
// Bu kurallar:
//  - Kullanıcıların yalnızca kendi user/farm dokümanlarını değiştirmesine izin verir
//  - market koleksiyonunda okumayı herkese açık tutar, yazma için authed olmayı şart koşar
//  - Karmaşık iş kuralları (coin/envanter kontrolü) client-side runTransaction ile yapılır;
//    kurallar kimliği koruyor, iş mantığı transaction atomikliğiyle.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // users/{uid}: yalnızca sahip yazabilir; okumayı kendine sınırla
    match /users/{uid} {
      allow read: if request.auth != null && (
        request.auth.uid == uid
        // İlan alımı için satıcı dokümanına da yazmak gerekir — aşağıdaki update kuralı bunu açar
      );
      allow create: if request.auth != null && request.auth.uid == uid;
      // Update: sahibi her şeyi; başkası yalnızca coin ve crops inventory alanlarını dokunabilir
      // (alım-satım transaction'larında karşı tarafın bakiyesini/inventasını değiştirmek için).
      allow update: if request.auth != null && (
        request.auth.uid == uid
        || request.resource.data.diff(resource.data).affectedKeys().hasOnly(['coins', 'inventory'])
      );
      allow delete: if false;
    }

    // farms/{uid}/tiles/** — yalnızca sahip okur ve yazar
    match /farms/{uid}/tiles/{tileId} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    // market — okuma herkese açık (giriş yapmış kullanıcılara), yazma authed
    match /market/{listingId} {
      allow read: if request.auth != null;
      // Create: sadece kendi adına ilan
      allow create: if request.auth != null
        && request.resource.data.sellerId == request.auth.uid
        && request.resource.data.quantity is int && request.resource.data.quantity >= 1
        && request.resource.data.pricePerUnit is int && request.resource.data.pricePerUnit >= 1;
      // Update: ilanlar update edilmez (alım silme yapar)
      allow update: if false;
      // Delete: satıcı kendi ilanını geri çekebilir, veya alan başkası silebilir
      allow delete: if request.auth != null;
    }
  }
}
